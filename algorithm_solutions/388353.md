# [지게차와 크레인](https://school.programmers.co.kr/learn/courses/30/lessons/388353)

### Solution

1. 첫 솔루션<br>
겉을 0으로 감싸고 bfs 사용
```
int solution(vector<string> storage, vector<string> requests) {

    int answer = storage.size()*storage[0].length();

    int dirx[4] = { 0,1,0,-1 };
    int diry[4] = { 1,0,-1,0 };

    storage.insert(storage.begin(), "00000000000000000000000000000000000000000000000000");
    storage.push_back("00000000000000000000000000000000000000000000000000");
    for (int i = 0; i < storage.size(); i++)
    {
        storage[i] = '0' + storage[i] + '0';
    }

    int n = storage.size();
    int m = storage[1].length();
    int s = n * m;

    bool* data = new bool[s];
    memset(data, false, s * sizeof(bool));

    bool** visit = new bool* [n];
    for (int i = 0; i < n; i++)
    {
        visit[i] = data + (i * m);
    }

    queue<pair<int, int>> q;
    set<pair<int, int>> rm;


    for(string req : requests)
    {
        // 지게차 사용
        if (req.length() == 1)
        {
            q.push({ 0,0 });

            while (!q.empty())
            {
                int y = q.front().first;
                int x = q.front().second;
                q.pop();

                if (visit[y][x] == true)
                    continue;

                visit[y][x] = true;

                for (int i = 0; i < 4; i++)
                {
                    int ny = y + diry[i];
                    int nx = x + dirx[i];
                    if (ny < 0 || ny >= n || nx < 0 || nx >= m)
                        continue;

                    if (storage[ny][nx] == req[0])
                    {
                        rm.insert({ ny,nx });
                    }
                    else if (storage[ny][nx] == '0')
                    {
                        q.push({ ny,nx });
                    }
                }
            }

            for (auto iter = rm.begin(); iter != rm.end(); iter++)
            {
                int y = iter->first;
                int x = iter->second;
                storage[y][x] = '0';
                answer--;
            }

            rm.clear();
            memset(data, false, s * sizeof(bool));
        }
        // 크레인 사용
        else 
        {
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < m; j++)
                {
                    if (storage[i][j] == req[0])
                    {
                        storage[i][j] = '0';
                        answer--;
                    }
                }
            }
        }
    }

    return answer;
}
```
