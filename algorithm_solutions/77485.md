# [행렬 테두리](https://school.programmers.co.kr/learn/courses/30/lessons/77485)

입력에 맞춰 필요한 범위의 데이터의 위치를 회전시킴

### Solution
```
vector<int> solution(int rows, int columns, vector<vector<int>> queries) {
    vector<int> answer;

    vector<vector<int>> arr(rows, vector<int>(columns));
    int idx = 1;
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < columns; j++)
        {
            arr[i][j] = idx++;
        }
    }


    for (int i = 0; i < queries.size(); i++)
    {
        int lx = queries[i][1] - 1;
        int ly = queries[i][0] - 1;
        int rx = queries[i][3] - 1;
        int ry = queries[i][2] - 1;

        int w = rx - lx;
        int h = ry - ly;

        int buffer = arr[ly][lx];
        int low = buffer;
        for (int i = 0; i < h; i++)
        {
            arr[ly+i][lx] = arr[ly+i+1][lx];
            low = min(low, arr[ly+i][lx]);
        }
        for (int i = 0; i < w; i++)
        {
            arr[ry][lx+i] = arr[ry][lx+i+1];
            low = min(low, arr[ry][lx + i]);
        }
        for (int i = 0; i < h; i++)
        {
            arr[ry-i][rx] = arr[ry-i-1][rx];
            low = min(low, arr[ry - i][rx]);
        }
        for (int i = 0; i < w - 1; i++)
        {
            arr[ly][rx-i] = arr[ly][rx-i-1];
            low = min(low, arr[ly][rx - i]);
        }
        arr[ly][lx+1] = buffer;
        answer.push_back(low);

    }
    return answer;
}
```
