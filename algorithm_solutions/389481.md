# [봉인된 주문](https://school.programmers.co.kr/learn/courses/30/lessons/389481)

### Solution

```c
string solution(long long n, vector<string> bans)
{
    string answer = "";
    set<long long> bansToIndex;


    //cache1 : 문자 길이 n 별로 만들 수 있는 조합
    //cache2 : 문자 길이 n 이 시작하는 인덱스
    long long cache1[12] = { 1,26 };
    long long cache2[12] = { 0,0 };

    for (size_t i = 2; i < 12; i++)
    {
        cache1[i] = cache1[i - 1] * 26;
        cache2[i] = cache2[i - 1] + cache1[i - 1];
    }

    // ban -> idx 변환
    for (string str : bans)
    {
        int len = str.length();
        long long idx = cache2[len];

        for (size_t i = 0; i < len; i++)
        {
            idx += (str[i] - 'a') * cache1[len - i - 1];
        }

        bansToIndex.insert(idx);
    }

    // n까지 가는 알파벳들 중 제거한 갯수 계산
    int count = 0;
    for (auto i = bansToIndex.begin(); i != bansToIndex.end(); i++)
    {
        if (n > *i)
        {
            n++;
            count++;
        }
        else break;
    }

    //n을 다시 단어로 변환
    int len = 0;
    for (size_t i = 11; i > 0; i--)
    {
        if (n >= cache2[i])
        {
            len = i;
            break;
        }
    }

    while (n > 0)
    {
        n--;
        char c = 'a' + (n % 26);
        answer = c + answer;
        n /= 26;
    }

    return answer;
}
```
